<?php
session_start();

// Database users array
$users = array(
    'admin' => array('password' => 'admin123', 'role' => 'Admin'),
    'editor' => array('password' => 'editor123', 'role' => 'Editor'),
    'viewer' => array('password' => 'viewer123', 'role' => 'Viewer')
);

// Define upload directory
$uploadDir = 'uploads/';

// Check if uploads directory exists, create if it doesn't
if (!is_dir($uploadDir)) {
    mkdir($uploadDir, 0755, true);
}

// Handle login form submission
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['username']) && isset($_POST['password'])) {
    $username = $_POST['username'];
    $password = $_POST['password'];
    
    if (isset($users[$username]) && $users[$username]['password'] == $password) {
        $_SESSION['username'] = $username;
        $_SESSION['role'] = $users[$username]['role'];
    } else {
        $errorMessage = "Invalid credentials. Login is required.";
    }
}

// Handle file upload
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_FILES['file']) && isset($_SESSION['username'])) {
    $userRole = $_SESSION['role'];
    if ($userRole == 'Admin' || $userRole == 'Editor') {
        $allowedTypes = array('docx', 'MP4', 'MP3');
        $uploadedFile = $_FILES['file'];
        $fileName = $uploadedFile['name'];
        $fileExtension = pathinfo($fileName, PATHINFO_EXTENSION);
        
        // Verify file types and extensions (mime and extension checking)
        $allowedMimes = array('application/vnd.openxmlformats-officedocument.wordprocessingml.document', 'video/mp4', 'audio/mpeg');
        $fileMime = $uploadedFile['type'];
        
        if (in_array($fileExtension, $allowedTypes) && in_array($fileMime, $allowedMimes)) {
            $targetPath = $uploadDir . $fileName;
            if (move_uploaded_file($uploadedFile['tmp_name'], $targetPath)) {
                $uploadMessage = "File uploaded successfully.";
            } else {
                $uploadError = "Failed to upload file.";
            }
        } else {
            $uploadError = "Invalid file type. Only docx, MP4, MP3 files are allowed.";
        }
    } else {
        $uploadError = "You do not have permission to upload files.";
    }
}

// Handle file deletion
if ($_SERVER['REQUEST_METHOD'] == 'POST' && isset($_POST['file_to_delete']) && isset($_SESSION['username'])) {
    $userRole = $_SESSION['role'];
    if ($userRole == 'Admin' || $userRole == 'Editor') {
        $fileToDelete = $_POST['file_to_delete'];
        $filePath = $uploadDir . $fileToDelete;
        if (file_exists($filePath)) {
            if (unlink($filePath)) {
                $deleteMessage = "File deleted successfully.";
            } else {
                $deleteError = "Failed to delete file.";
            }
        } else {
            $deleteError = "File does not exist.";
        }
    } else {
        $deleteError = "You do not have permission to delete files.";
    }
}

// Handle logout
if (isset($_GET['logout'])) {
    session_destroy();
    header("Location: what3.php");
    exit();
}

// Get list of uploaded files
function listUploadedFiles($directory) {
    $files = array();
    if (is_dir($directory)) {
        $fileList = scandir($directory);
        foreach ($fileList as $file) {
            if ($file != '.' && $file != '..') {
                $files[] = $file;
            }
        }
    }
    return $files;
}
