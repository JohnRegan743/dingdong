<?php
session_start();

$users = array(
    'admin' => array('password' => 'admin123', 'role' => 'Admin'),
    'editor' => array('password' => 'editor123', 'role' => 'Editor'),
    'viewer' => array('password' => 'viewer123', 'role' => 'Viewer')
);

$permissions = array(
    'Admin' => array('upload' => true, 'edit' => true, 'delete' => true, 'read' => true),
    'Editor' => array('upload' => false, 'edit' => true, 'delete' => false, 'read' => true),
    'Viewer' => array('upload' => false, 'edit' => false, 'delete' => false, 'read' => true)
);

// Define upload directory
$upload_dir = 'uploads/';

// Create upload directory if it doesn't exist
if (!is_dir($upload_dir)) {
    mkdir($upload_dir, 0755, true);
}

// Handle login
if (isset($_POST['username']) && isset($_POST['password'])) {
    $username = $_POST['username'];
    $password = $_POST['password'];
    
    if (isset($users[$username]) && $users[$username]['password'] == $password) {
        $_SESSION['username'] = $username;
        $_SESSION['role'] = $users[$username]['role'];
    } else {
        $error_message = "Invalid credentials!";
    }
}

// Handle file upload
if (isset($_POST['upload']) && isset($_SESSION['role']) && $permissions[$_SESSION['role']]['upload']) {
    if (isset($_FILES['file']) && $_FILES['file']['error'] == 0) {
        $allowed_extensions = array('txt', 'jpg', 'png', 'pdf');
        $file_extension = strtolower(pathinfo($_FILES['file']['name'], PATHINFO_EXTENSION));
        $file_size = $_FILES['file']['size'];
        
        if (in_array($file_extension, $allowed_extensions) && $file_size <= 16777216) {
            $filename = basename($_FILES['file']['name']);
            $target_path = $upload_dir . $filename;
            
            if (move_uploaded_file($_FILES['file']['tmp_name'], $target_path)) {
                $success_message = "File uploaded successfully!";
            } else {
                $error_message = "Failed to upload file!";
            }
        } else {
            $error_message = "Invalid file type or size exceeds 16MB!";
        }
    } else {
        $error_message = "No file selected or upload error!";
    }
}

// Handle file deletion
if (isset($_POST['delete']) && isset($_POST['filename']) && isset($_SESSION['role'])) {
    if ($permissions[$_SESSION['role']]['delete']) {
        $filename = basename($_POST['filename']); // Security: prevent directory traversal
        $file_path = $upload_dir . $filename;
        
        if (file_exists($file_path) && is_file($file_path)) {
            if (unlink($file_path)) {
                $success_message = "File deleted successfully!";
            } else {
                $error_message = "Failed to delete file!";
            }
        } else {
            $error_message = "File not found!";
        }
    } else {
        $error_message = "You don't have permission to delete files!";
    }
}

// Handle logout
if (isset($_POST['logout'])) {
    session_destroy();
    header('Location: ' . $_SERVER['PHP_SELF']);
    exit;
}

?>
<!DOCTYPE html>
<html>
<head>
    <title>File Manager</title>
</head>
<body>
    <h2>File Manager</h2>
    
    <?php
    // Display error or success messages
    if (isset($error_message)) {
        echo "<div class='error'>$error_message</div>";
    }
    if (isset($success_message)) {
        echo "<div class='success'>$success_message</div>";
    }
    ?>

    <?php if (!isset($_SESSION['username'])): ?>
        <!-- Login Form -->
        <form method="POST">
            <label>Username: <input type="text" name="username" required></label><br>
            <label>Password: <input type="password" name="password" required></label><br>
            <input type="submit" value="Login">
        </form>
    <?php else: ?>
        <p>Welcome, <?php echo htmlspecialchars($_SESSION['username']); ?>! 
           Role: <?php echo htmlspecialchars($_SESSION['role']); ?></p>
        
        <?php if ($permissions[$_SESSION['role']]['upload']): ?>
            <!-- File Upload Form -->
            <h3>Upload File</h3>
            <form method="POST" enctype="multipart/form-data">
                <label>Select File: <input type="file" name="file" required></label>
                <input type="submit" name="upload" value="Upload">
            </form>
            <p><small>Allowed types: txt, jpg, png, pdf (Max size: 16MB)</small></p>
        <?php endif; ?>
        
        <!-- File List -->
        <h3>Files</h3>
        <?php
        if (is_dir($upload_dir)) {
            $files = scandir($upload_dir);
            $files = array_diff($files, array('.', '..'));
            
            if (!empty($files)) {
                echo "<table><tr><th>File Name</th><th>Size</th><th>Actions</th></tr>";
                foreach ($files as $file) {
                    $file_path = $upload_dir . $file;
                    if (is_file($file_path)) {
                        $file_size = filesize($file_path);
                        $file_size_formatted = number_format($file_size / 1024, 2) . ' KB';
                        
                        echo "<tr>";
                        echo "<td>" . htmlspecialchars($file) . "</td>";
                        echo "<td>$file_size_formatted</td>";
                        echo "<td>";
                        
                        // Download link
                        echo "<a href='$file_path' target='_blank'>View/Download</a>";
                        
                        // Delete button (if user has permission)
                        if ($permissions[$_SESSION['role']]['delete']) {
                            echo " | ";
                            echo "<form method='POST' style='display:inline;'>";
                            echo "<input type='hidden' name='filename' value='" . htmlspecialchars($file) . "'>";
                            echo "<input type='submit' name='delete' value='Delete' onclick='return confirm(\"Are you sure you want to delete this file?\");'>";
                            echo "</form>";
                        }
                        
                        echo "</td>";
                        echo "</tr>";
                    }
                }
                echo "</table>";
            } else {
                echo "<p>No files available.</p>";
            }
        } else {
            echo "<p>Upload directory not accessible.</p>";
        }
        ?>
        
        <!-- Logout Form -->
        <form method="POST" style="margin-top: 20px;">
            <input type="submit" name="logout" value="Logout">
        </form>
    <?php endif; ?>
</body>
</html>
